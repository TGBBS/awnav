{{- $item := .item -}} {{- $site := .site -}} {{- $class := .class | default "" -}} {{- $width := .width | default 40 -}} {{- $height := .height | default 40 -}} {{- $loading := .loading | default "lazy" -}} {{- $fetchpriority := .fetchpriority | default "low" -}} {{- $logo := $item.logo | default "" -}} {{- $itemURL := $item.url | default "" -}} {{- $defaultLogo := "" -}} {{- if $site.Params.defaultLogo -}} {{- $defaultLogo = $site.Params.defaultLogo | relURL -}} {{- end -}} {{- $resolvedLogo := "" -}} {{- if $logo -}} {{- if findRE "http.*?(.|\n)*?" $logo -}} {{- $resolvedLogo = $logo -}} {{- else if hasPrefix $logo "/" -}} {{- $resolvedLogo = $logo | relURL -}} {{- else -}} {{- $resolvedLogo = printf "%s/%s" $site.Params.logosPath $logo | relURL -}} {{- end -}} {{- end -}} {{- $thumbLogo := "" -}} {{- if and $resolvedLogo (hasPrefix $resolvedLogo "/logos/") -}} {{- $thumbLogo = printf "/logos-thumb/%s.webp" (strings.TrimPrefix "/logos/" $resolvedLogo) | relURL -}} {{- end -}} {{- $telegramUsername := "" -}} {{- if gt (len (findRE `(?i)^https?://(?:t|telegram)\.me/[^/?#]+/?` $itemURL)) 0 -}} {{- $telegramUsername = replaceRE `(?i)^https?://(?:t|telegram)\.me/([^/?#]+).*$` "$1" $itemURL -}} {{- end -}} {{- $canUseTelegramAvatar := and $telegramUsername (ne (lower $telegramUsername) "joinchat") (not (hasPrefix $telegramUsername "+")) (ne (lower $telegramUsername) "c") -}} {{- $telegramAvatar := "" -}} {{- if $canUseTelegramAvatar -}} {{- $telegramAvatar = printf "https://t.me/i/userpic/320/%s.jpg" $telegramUsername -}} {{- end -}} {{- $primarySrc := $resolvedLogo -}} {{- $fallbackSrc := $defaultLogo -}} {{- if $thumbLogo -}} {{- $primarySrc = $thumbLogo -}} {{- if $resolvedLogo -}} {{- $fallbackSrc = $resolvedLogo -}} {{- end -}} {{- end -}} {{- if not $primarySrc -}} {{- $primarySrc = $telegramAvatar -}} {{- end -}} {{- if not $primarySrc -}} {{- $primarySrc = $defaultLogo -}} {{- end -}}
